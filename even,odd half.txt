#include <stdio.h>

void divideArray(int nums[], int size) {
    int i, j = 0, k = size - 1;
    int result[size];

    for (i = 0; i < size; i++) {
        if (nums[i] % 2 == 0) {
            result[j] = nums[i];
            j++;
        } else {
            result[k] = nums[i];
            k--;
        }
    }

    printf("Even Half: ");
    for (i = 0; i < size / 2; i++) {
        printf("%d ", result[i]);
    }

    printf("\nOdd Half: ");
    for (i = size / 2; i < size; i++) {
        printf("%d ", result[i]);
    }
}

int main() {
    int nums[] = {1, 2, 3, 4, 5, 6};
    int size = sizeof(nums) / sizeof(nums[0]);

    divideArray(nums, size);

    return 0;
}
