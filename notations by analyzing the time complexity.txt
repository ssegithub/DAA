#include <stdio.h>

int main() {
   
    int n = 10;
    for (int i = 0; i < n; i++) {
        printf("Hello, World!\n");
    }


    int m = 5;
    for (int j = 0; j < m; j++) {
        printf("Welcome to C Programming!\n");
    }


    int k = 8;
    for (int l = 0; l < k; l++) {
        printf("This is a Theta Notation example.\n");
    }

    return 0;
}
